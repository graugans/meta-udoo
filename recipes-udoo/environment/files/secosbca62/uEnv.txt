# HDMI
video=fbmem=24M video=mxcfb0:dev=hdmi,1920x1080M@60,bpp=32

# 7" LVDS Panel
#video=fbmem=24M video=mxcfb0:dev=ldb,LDB-WVGA,if=RGB666,bpp=32
